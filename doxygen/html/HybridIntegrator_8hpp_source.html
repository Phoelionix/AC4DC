58 template<typename T>
59 void Hybrid<T>::iterate(double t_initial, double t_final) {
60 
61  if (this->dt < 1E-16) {
62  std::cerr<<"WARN: step size "<<this->dt<<"is smaller than machine precision"<<std::endl;
63  } else if (this->dt < 0) {
64  throw std::runtime_error("Step size is negative!");
65  }
66 
67  size_t npoints = (t_final - t_initial)/this->dt + 1;
68 
69  npoints = (npoints >= this->order) ? npoints : this->order;
70  // Set up the containters
71  this->t.resize(npoints);
72  this->y.resize(npoints);
73 
74  // Set up the t grid
75  this->t[0] = t_initial;
76 
77  for (size_t n=1; n<npoints; n++){
78  this->t[n] = this->t[n-1] + this->dt;
79  }
80  this->run_steps();
81 }

109 template<typename T>
110 void Hybrid<T>::Moulton(unsigned n){
111  T old = this->y[n];
112  old *= -1.;
113  old += this->y[n+1];
114 
115  // Adams-Moulton step
116  // Here, tmp is the y_n+1 guessed in the preceding step, handle i=0 explicitly:
117  T tmp;
118  tmp *= 0;
119  // Now tmp goes back to being an aggregator
120  for (int i = 1; i < this->order; i++) {
121  T ydot;
122  this->sys2(this->y[1+n-i], ydot, this->t[1+n-i]);
123  ydot *= this->b_AM[i];
124  tmp += ydot;
125  }
126 
127  //this->y[n+1] = this->y[n] + tmp * this->dt;
128  tmp *= this->dt;
129  tmp += this->y[n];
130 
131  // tmp now stores the additive constant for y_n+1 = tmp + h b0 f(y_n+1, t_n+1)
132  T prev;
133  double diff = stiff_rtol*2;
134  unsigned idx=0;
135  while (diff > stiff_rtol && idx < stiff_max_iter){
136  // solve by Picard iteration
137  prev = this->y[n+1];
138  prev *= -1;
139  T dydt;
140  this->sys2(this->y[n+1], dydt, this->t[n+1]);
141  dydt *= this->b_AM[0]*this->dt;
142  this->y[n+1] = tmp;
143  this->y[n+1] += dydt;
144  prev += this->y[n+1];
145  diff = prev.norm()/this->y[n+1].norm();
146  idx++;
147  }
148  if(idx==stiff_max_iter) std::cerr<<"Max Euler iterations exceeded, err = "<<diff<<std::endl;
149  this->y[n+1] += old;
150 }

152 template<typename T>
153 void Hybrid<T>::backward_Euler(unsigned n){
154  // Assumes that y_n+1 contains a guess based on sys, and estimates a solution to y_n
155  unsigned idx = 0;
156  T old = this->y[n];
157  old *= -1.;
158  old += this->y[n+1];
159  //old caches the step from the regular part
160 
161  // Guess. (Naive Euler)
162  sys2(this->y[n+1], this->y[n+1], this->t[n+1]);
163  this->y[n+1] *= this->dt;
164  this->y[n+1] += this->y[n];
165 
166  double diff = stiff_rtol*2;
167  while (diff > stiff_rtol && idx < stiff_max_iter){
168  T tmp = this->y[n+1];
169  sys2(tmp, this->y[n+1], this->t[n+1]);
170  this->y[n+1] *= this->dt;
171  this->y[n+1] += this->y[n];
172 
173  tmp *= -1;
174  tmp += this->y[n+1];
175  diff = tmp.norm()/this->y[n].norm();
176  idx++;
177  }
178  this->y[n+1] += old;
179  if(idx==stiff_max_iter) std::cerr<<"Max Euler iterations exceeded"<<std::endl;
180 }
