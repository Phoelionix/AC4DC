158 template<typename T>
159 void Adams_BM<T>::step(int n) {
160  // Predicts the value y_n+1
161  // Adams-Bashforth predictor routine:
162 
163  T tmp, ydot;
164  tmp = this->zero_y;
165 
166  for (size_t i = 0; i < order; i++) {
167  this->sys(this->y[n-i], ydot, this->t[n-i]);
168  ydot *= b_AB[i];
169  tmp += ydot;
170  }
171 
172  // Weird syntax here is done in order to avoid calls to
173  // operator+ and orparator*, which may unnecessarily create large
174  // local variables.
175 
176  // y_n+1 = y_n + dt*sum_{i=0}^s-1 b_i f_(n-i)
177  tmp *= this->dt;
178  tmp += this->y[n];
179 
180  // Adams-Moulton corrector step
181  // Here, tmp is the y_n+1 guessed in the preceding step, handle i=0 explicitly:
182  T tmp2(tmp);
183  this->sys(tmp2, tmp, this->t[n+1]);
184  tmp *= b_AM[0];
185  // Now tmp goes back to being an aggregator
186  for (int i = 1; i < order; i++) {
187  this->sys(this->y[n-i+1], ydot, this->t[n-i+1]);
188  ydot *= b_AM[i];
189  tmp += ydot;
190  }
191  // Store the corrected value
192  //this->y[n+1] = this->y[n] + tmp * this->dt;
193  tmp *= this->dt;
194  tmp += this->y[n];
195  this->y[n+1] = tmp;
196 }

198 template<typename T>
199 void Adams_BM<T>::iterate(double t_initial, double t_final) {
200 
201  if (this->dt < 1E-16) {
202  std::cerr<<"WARN: step size "<<this->dt<<"is smaller than machine precision"<<std::endl;
203  } else if (this->dt < 0) {
204  throw std::runtime_error("Step size is negative!");
205  }
206 
207  size_t npoints = (t_final - t_initial)/this->dt + 1;
208 
209  npoints = (npoints >= order) ? npoints : order;
210  // Set up the containters
211  this->t.resize(npoints);
212  this->y.resize(npoints);
213 
214  // Set up the t grid
215  this->t[0] = t_initial;
216 
217  for (int n=1; n<npoints; n++){
218  this->t[n] = this->t[n-1] + this->dt;
219  }
220  this->run_steps();
221 }
